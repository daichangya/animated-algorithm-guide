    public static int binarySearch(int[] arr, int target) {
        int left = 0, right = arr.length - 1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;  // é˜²æ­¢æº¢å‡º
            
            if (arr[mid] == target) {
                return mid;

    public static int lowerBound(int[] arr, int target) {

        while (left < right) {

                right = mid;

        }
        
        return left;
    }
